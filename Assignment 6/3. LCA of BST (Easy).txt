#include "bits/stdc++.h"

using namespace std;
#define debug(...) cout<< " [" << #__VA_ARGS__ ": " << (__VA_ARGS__) << "] " ;
typedef long long int ll;
const ll mod = 1e9 + 7;
class tree {
public:
    int data;
    tree *left, *right;

    tree(int x) {
        this->data = x;
        this->left = this->right = NULL;
    }
};

tree *build(int index, tree *root, int arr[], int n) {
    if (index < n) {
        if (arr[index] == -1) return NULL; //-1 denotes NULL
        tree *temp = new tree(arr[index]);
        root = temp;
        root->left = build(2 * index + 1, root->left, arr, n);
        root->right = build(2 * index + 2, root->right, arr, n);
    }
    return root;
}
tree* lca(tree *root, int n1, int n2){
    if(root==NULL)  return NULL;
    if(n1 <=root->data && root->data<=n2)   return root;
    if(root->data >=n1 && root->data >=n2)  return lca(root->left,n1,n2);
    if(root->data <=n1 && root->data <=n2)  return lca(root->right,n1,n2);
}
int main() {
    int n,n1,n2;
    cin>>n;
